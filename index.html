/** * Draw the wheel (5 sectors) with nicer gradients + outer-ring labels. * - labels placed on outer ring so they never get blocked by center Spin button * - supports any number of sectors (not only 5) * * Usage: * const canvas = document.getElementById("wheel"); * const ctx = canvas.getContext("2d"); * drawWheel(ctx, canvas, sectors); * * sectors example: * [{ label:"Blessing" }, { label:"This is on us" }, ...] */ function drawWheel(ctx, canvas, sectors) { if (!ctx || !canvas) return; if (!Array.isArray(sectors) || sectors.length === 0) return; // --- HiDPI crisp --- const cssSize = canvas.clientWidth || 420; const dpr = window.devicePixelRatio || 1; canvas.width = Math.floor(cssSize * dpr); canvas.height = Math.floor(cssSize * dpr); canvas.style.width = cssSize + "px"; canvas.style.height = cssSize + "px"; ctx.setTransform(dpr, 0, 0, dpr, 0, 0); const w = cssSize; const h = cssSize; const cx = w / 2; const cy = h / 2; const radius = Math.min(w, h) / 2; ctx.clearRect(0, 0, w, h); // --- Style tuning (you can tweak) --- const OUTER_R = radius * 0.98; const INNER_CLEAR_R = radius * 0.38; // keeps center clean for Spin button const LABEL_R = radius * 0.78; // label ring radius (outer) const LABEL_FONT = "600 18px ui-serif, Georgia, 'Times New Roman', serif"; const LABEL_COLOR = "rgba(60,48,40,0.9)"; const BORDER_COLOR = "rgba(120, 100, 85, 0.25)"; // 5 “base” colors (cycled if more than 5 sectors) const baseColors = [ { a: "#F6E7A9", b: "#FDF4D5" }, // warm yellow { a: "#F6C9D3", b: "#FCE6EC" }, // soft pink { a: "#D7E7FF", b: "#EEF5FF" }, // soft blue { a: "#D6F2E7", b: "#EFFFF8" }, // mint { a: "#F6DFC5", b: "#FFF1E3" }, // peach ]; // --- geometry --- const n = sectors.length; const arc = (Math.PI * 2) / n; // Rotate so sector #0 is centered at top under pointer // (Pointer at 12 o'clock. Mid of first sector at top) const startOffset = -Math.PI / 2 - arc / 2; // --- helpers --- const clamp = (v, a, b) => Math.max(a, Math.min(b, v)); function shortenTextToFit(text, maxPx) { // conservative shortening by measuring const t = String(text || "").trim(); if (!t) return ""; if (ctx.measureText(t).width <= maxPx) return t; const ell = "…"; let lo = 0, hi = t.length; while (lo < hi) { const mid = Math.ceil((lo + hi) / 2); const s = t.slice(0, mid) + ell; if (ctx.measureText(s).width <= maxPx) lo = mid; else hi = mid - 1; } return t.slice(0, lo) + ell; } function drawOuterBevel() { // subtle outer rim (premium feel) ctx.save(); ctx.beginPath(); ctx.arc(cx, cy, OUTER_R, 0, Math.PI * 2); ctx.closePath(); const rimGrad = ctx.createRadialGradient(cx, cy, OUTER_R * 0.72, cx, cy, OUTER_R); rimGrad.addColorStop(0, "rgba(255,255,255,0.35)"); rimGrad.addColorStop(0.7, "rgba(255,255,255,0.10)"); rimGrad.addColorStop(1, "rgba(110,90,70,0.20)"); ctx.strokeStyle = rimGrad; ctx.lineWidth = radius * 0.035; ctx.stroke(); // inner thin rim line ctx.beginPath(); ctx.arc(cx, cy, OUTER_R * 0.93, 0, Math.PI * 2); ctx.strokeStyle = "rgba(120,100,85,0.18)"; ctx.lineWidth = 1.5; ctx.stroke(); ctx.restore(); } function fillSector(startAngle, endAngle, colorA, colorB) { ctx.save(); ctx.beginPath(); ctx.moveTo(cx, cy); ctx.arc(cx, cy, OUTER_R, startAngle, endAngle); ctx.closePath(); // sector gradient: radial + slightly directional const g = ctx.createRadialGradient(cx, cy, OUTER_R * 0.2, cx, cy, OUTER_R); g.addColorStop(0, colorB); g.addColorStop(1, colorA); ctx.fillStyle = g; ctx.fill(); // separator line ctx.strokeStyle = BORDER_COLOR; ctx.lineWidth = 1.5; ctx.stroke(); ctx.restore(); } function drawLabelOnOuterRing(startAngle, endAngle, text) { const mid = (startAngle + endAngle) / 2; ctx.save(); ctx.translate(cx, cy); ctx.rotate(mid); // Make text readable (flip when on left side) // After rotate(mid), "up" is -y. const shouldFlip = mid > Math.PI / 2 && mid < Math.PI * 1.5; if (shouldFlip) ctx.rotate(Math.PI); ctx.font = LABEL_FONT; ctx.fillStyle = LABEL_COLOR; ctx.textAlign = "center"; ctx.textBaseline = "middle"; // max width allowed for this sector at label radius const sectorArcLen = arc * LABEL_R; const maxTextWidth = clamp(sectorArcLen * 0.85, 60, 260); const label = shortenTextToFit(text, maxTextWidth); // draw the text on outer ring, slightly inside ctx.fillText(label, 0, -LABEL_R); ctx.restore(); } function clearCenterArea() { // creates a clean center (Spin button area) ctx.save(); ctx.beginPath(); ctx.arc(cx, cy, INNER_CLEAR_R, 0, Math.PI * 2); ctx.closePath(); const centerGrad = ctx.createRadialGradient(cx, cy, INNER_CLEAR_R * 0.15, cx, cy, INNER_CLEAR_R); centerGrad.addColorStop(0, "rgba(255,255,255,0.15)"); centerGrad.addColorStop(1, "rgba(255,255,255,0.38)"); ctx.fillStyle = centerGrad; ctx.fill(); ctx.strokeStyle = "rgba(120,100,85,0.18)"; ctx.lineWidth = 2; ctx.stroke(); ctx.restore(); } // --- Draw sectors --- for (let i = 0; i < n; i++) { const start = startOffset + i * arc; const end = start + arc; const c = baseColors[i % baseColors.length]; fillSector(start, end, c.a, c.b); } // --- Outer rim --- drawOuterBevel(); // --- Labels on outer ring --- for (let i = 0; i < n; i++) { const start = startOffset + i * arc; const end = start + arc; drawLabelOnOuterRing(start, end, sectors[i]?.label ?? ""); } // --- Center clean area --- clearCenterArea(); }
